function drawPlayer(){ if(player.invincibleTime>0 && Math.floor(Date.now()/120)%2==0) return; const cx = player.x + 25; const cy = player.y + 30; let bodyColor = getPlayerColor(); let glowIntensity = Math.min(30, 10 + playerLevel * 2); ctx.shadowBlur = 0; ctx.shadowColor = bodyColor; ctx.shadowBlur = glowIntensity; // 主战机 function drawShip(x, y, scale){ ctx.beginPath(); ctx.moveTo(x + 25*scale, y); ctx.lineTo(x + 10*scale, y + 40*scale); ctx.lineTo(x + 15*scale, y + 55*scale); ctx.lineTo(x + 25*scale, y + 60*scale); ctx.lineTo(x + 35*scale, y + 55*scale); ctx.lineTo(x + 40*scale, y + 40*scale); ctx.closePath(); ctx.fillStyle = bodyColor; ctx.fill(); } // 等级外观：1 → 2 → 3 → 4 架 const stage = Math.min(4, Math.floor(playerLevel / 2) + 1); // 每2级升一个外观阶段 // 1架：中间 drawShip(player.x, player.y, 1); // 2架：左右 if (stage >= 2) { drawShip(player.x - 50, player.y + 12, 0.7); drawShip(player.x + 50, player.y + 12, 0.7); } // 3架：左右+前翼 if (stage >= 3) { drawShip(player.x - 32, player.y - 8, 0.6); drawShip(player.x + 32, player.y - 8, 0.6); } // 4架：四机菱形编队（最帅形态） if (stage >= 4) { drawShip(player.x - 66, player.y + 20, 0.55); drawShip(player.x + 66, player.y + 20, 0.55); drawShip(player.x - 20, player.y - 16, 0.5); drawShip(player.x + 20, player.y - 16, 0.5); } // 护盾 if(player.shieldCount > 0){ let opacity = player.shieldCount / player.shieldMax; ctx.strokeStyle = bodyColor; ctx.lineWidth=2; ctx.globalAlpha = opacity; ctx.beginPath(); ctx.arc(cx, cy, 40, 0, Math.PI*2); ctx.stroke(); ctx.strokeStyle = '#fff'; ctx.lineWidth=1; ctx.beginPath(); ctx.arc(cx, cy, 36, 0, Math.PI*2); ctx.stroke(); ctx.globalAlpha = 1; } ctx.shadowBlur = 0; }
